<?php namespace App\Http\Controllers; use App\Models\Ticket; use Illuminate\Http\Request; use Illuminate\Http\RedirectResponse; use Illuminate\View\View; use Inertia\Inertia; class TicketController extends Controller { public function index(Request $request) { $query = Ticket::with(['creator', 'assignee']); // TicketFile Model <?php namespace App\Models; use Illuminate\Database\Eloquent\Factories\HasFactory; use Illuminate\Database\Eloquent\Model; use Illuminate\Database\Eloquent\Relations\BelongsTo; class TicketFile extends Model { use HasFactory; protected $fillable = [ 'ticket_id', 'user_id', 'filename', 'path', 'size', 'mime_type', ]; // Relationships public function ticket(): BelongsTo { return $this->belongsTo(Ticket::class); } public function user(): BelongsTo { return $this->belongsTo(User::class); } // Accessors public function getFileUrlAttribute(): string { return asset('storage/ticket-files/' . $this->path); } public function getFileSizeFormattedAttribute(): string { $bytes = $this->size; $units = ['B', 'KB', 'MB', 'GB']; for ($i = 0; $bytes > 1024 && $i < count($units) - 1; $i++) { $bytes /= 1024; } return round($bytes, 2) .
